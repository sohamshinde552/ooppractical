import java.util.*;


class ArrayOp {
    // Display the array
    void display(int[] arr) {
        for (int i = 0; i < arr.length; i++) {
            System.out.println("arr[" + i + "] = " + arr[i]);
        }
    }


    // Calculate the sum of elements in the array and average
    void sum(int[] arr) {
        int sum = 0;
        for (int i = 0; i < arr.length; i++) {
            sum = sum + arr[i];
        }
        System.out.println("Sum of all the elements in the array is: " + sum);
        int average = sum / arr.length;
        System.out.println("Average of elements in the array is: " + average);
    }


    // Find the maximum element in the array
    void maximum(int[] arr) {
        int max = arr[0];
        for (int i = 1; i < arr.length; i++) {
            if (max < arr[i]) {
                max = arr[i];
            }
        }
        System.out.println("Maximum: " + max);
    }


    // find minimum element
    void minimum(int[] arr) {
        int min = arr[0];
        for (int i = 1; i < arr.length; i++) {
            if (min > arr[i]) {
                min = arr[i];
            }
        }
        System.out.println("Minimum: " + min);
    }


    // Search an element
    void linearSearch(int[] arr, int key) {
        for (int i = 0; i < arr.length; i++) {
            if (key == arr[i]) {
                System.out.println("Element found at index : " + i);
            }
        }
    }


}

Assignment No. 6 - Interface In Java


package Assignments;


import java.util.Scanner;


interface Interest {
    double calculate(double p, double r, double t);
}


class SimpleInterest implements Interest {
    public double calculate(double p, double r, double t) {
        return (p * r * t) / 100;
    }
}


class CompoundInterest implements Interest {
    public double calculate(double p, double r, double t) {
        return p * Math.pow((1 + r / 100), t) - p;
    }
}


public class Assignment6 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);


        System.out.print("Enter Principal: ");
        double p = sc.nextDouble();


        System.out.print("Enter Rate: ");
        double r = sc.nextDouble();


        System.out.print("Enter Time: ");
        double t = sc.nextDouble();


        Interest si = new SimpleInterest();
        Interest ci = new CompoundInterest();


        System.out.println("Simple Interest: " + si.calculate(p, r, t));
        System.out.println("Compound Interest: " + ci.calculate(p, r, t));


        sc.close();
    }
}

Assignment No. 7 - ATM machine




import java.util.*;


class ATM {
    public static double balance = 50000;
    public static int pass = 1234;


    static void checkBalance(int pin) {
        if (pass == pin) {
            System.out.println("Balance: " + balance + " ₹\n");
        } else {
            System.out.println("Incorrect PIN\n");
        }
    }


    static void withdraw(int pin, double amount) {
        if (pass != pin) {
            System.out.println("Incorrect PIN\n");
            return;
        }


        if (amount <= 0) {
            System.out.println("Enter valid withdrawal amount\n");
            return;
        }


        if (amount > balance) {
            System.out.println("Insufficient Balance\n");
            return;
        }


        balance = balance - amount;
        System.out.println(amount + " ₹ Deducted from the account\n");
    }


    static void deposite(int pin, double amount) {
        if (pass != pin) {
            System.out.println("Incorrect PIN\n");
            return;
        }


        if (amount <= 0) {
            System.out.println("Enter valid deposit amount\n");
            return;
        }


        balance = balance + amount;
        System.out.println(amount + " ₹ Deposited in the account\n");
    }
}

Assignment No. 8.1 - Multithreaded Chat system in java


public class Assignment8 {
    static class ChatUser extends Thread {
        private String userName;


        public ChatUser(String name, int priority) {
            this.userName = name;
            setName(name); // Assign thread name
            setPriority(priority); // Assign priority
        }


        public void run() {
            try {
                for (int i = 1; i < 3; i++) {
                    System.out.println(getName() + " is sending message " + i);
                    Thread.sleep(500);


                    System.out.println(getName() + " received reply " + i);
                    Thread.sleep(500);
                }
            } catch (InterruptedException e) {
                System.out.println(getName() + " interrupted");
            }
        }
    }


    public static class ChatApp {
        public static void main(String[] args) {
            ChatUser user1 = new ChatUser("Alice", Thread.MAX_PRIORITY);
            ChatUser user2 = new ChatUser("Bob", Thread.MIN_PRIORITY);


            // Start threads
            user1.start();
            user2.start();


            // 4. Check thread status
            System.out.println("Is Alice alive? " + user1.isAlive());
            System.out.println("Is Bob alive? " + user2.isAlive());


            try {
                // 5. Synchronize using join()
                user1.join(); // main waits for Alice
                user2.join(); // then waits for Bob
            } catch (InterruptedException e) {
                e.printStackTrace();
            }


            System.out.println("Both chats finished");


            // 6. Demonstration of interruption instead of deprecated stop/suspend/resume
            try {
                ChatUser user3 = new ChatUser("Charlie", Thread.NORM_PRIORITY);
                user3.start();


                Thread.sleep(1000);


                user3.interrupt();
                System.out.println("Charlie interrupted");


                user3.join();
                System.out.println("Charlie finished");
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
        }
    }
}

                                    rs.getString("course"));
                        }


                    }

                    }
                }
            }


        } catch (Exception e) {
